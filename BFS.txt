from collections import deque

def bfs(graph, start):
    visited = set()
    queue = deque([start])

    while queue:
        node = queue.popleft()
        if node not in visited:
            print(node)
            visited.add(node)
            neighbors = graph[node]
            queue.extend(neighbors)

# Example usage
graph = {
    '1': ['2', '3'],
    '2': ['4'],
    '3': ['4', '7'],
    '4': ['5', '6'],
    '7': [],
    '5': [],
    '6': []
}

bfs(graph, '1')
